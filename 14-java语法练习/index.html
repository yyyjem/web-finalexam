// 新闻数据分析处理器
public class NewsAnalyzer {
    private List<Article> articles; // 文章列表

    // 构造函数
    public NewsAnalyzer(List<Article> articles) {
        this.articles = articles;
    }

    /**
     * 统计关键词出现频率
     * @return 关键词-数量的映射表
     */
    public Map<String, Integer> countKeywords() {
        return articles.stream()
            .flatMap(a -> Arrays.stream(a.getKeywords()))
            .collect(Collectors.groupingBy(
                keyword -> keyword,
                Collectors.summingInt(k -> 1)
            ));
    }
}
